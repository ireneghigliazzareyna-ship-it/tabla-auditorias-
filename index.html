// 1. Lista oficial de Auditores Activos y sus nombres legibles const AUDITORES_ACTIVOS = [ { nombre: "Araceli Chavira", keywords: ["araceli", "chavira"] }, { nombre: "Angelica Garcia", keywords: ["angelica", "garcia"] }, { nombre: "Adonnay Ramirez", keywords: ["adonnay", "ramirez"] }, { nombre: "Alejandra Torres", keywords: ["alejandra", "torres"] }, { nombre: "Montserrat Covarrubias", keywords: ["montserrat", "covarrubias"] }, { nombre: "Irene Ghigliazza", keywords: ["irene", "ghigliazza"] }, { nombre: "Brenda Lira", keywords: ["brenda", "lira"] }, { nombre: "Blanca Esther", keywords: ["blanca", "martinez", "esther"] }, { nombre: "Andy Alois", keywords: ["andy", "alois"] } ]; // Función para mapear el correo al nombre exacto o descartarlo si no es activo function obtenerAuditorActivo(correo) { if (!correo) return null; const correoLower = correo.toLowerCase(); for (const auditor of AUDITORES_ACTIVOS) { // Coincidencia si el correo contiene alguna palabra clave del auditor const match = auditor.keywords.some(key => correoLower.includes(key)); if (match) { return auditor.nombre; } } return null; // Si no coincide con ninguno, se considera inactivo } // Función principal que procesa la respuesta del query function procesarDatosDashboard(datosRaw) { // 2. Filtrar solo los registros de auditores activos y mapear nombres const datosFiltrados = []; datosRaw.forEach(item => { const nombreValido = obtenerAuditorActivo(item.AUDITOR); // Si el auditor coincide con la lista activa, lo conservamos if (nombreValido) { datosFiltrados.push({ ...item, AUDITOR: nombreValido // Reemplaza el correo por el nombre limpio }); } }); // 3. Renderizar la tabla y poblar los dropdowns renderTabla(datosFiltrados); poblarFiltroProcesos(datosFiltrados); poblarFiltroAuditores(); actualizarKpis(datosFiltrados); } // Render en la tabla (Grid) function renderTabla(datos) { const tbody = document.getElementById('grid-body'); // Ajusta al ID de tu tbody.innerHTML = ''; datos.forEach(row => { const tr = document.createElement('tr'); tr.innerHTML = ` ${row.FECHA_HORA ? new Date(row.FECHA_HORA.value || row.FECHA_HORA).toLocaleString() : '-'} ${row.PROCESO || '-'} ${row.AUDITOR} ${row.ID || '-'} `; tbody.appendChild(tr); }); } // Llenar Dropdown de Procesos dinámicamente con todos los que vengan en el query (TMWH HU, OUT, etc.) function poblarFiltroProcesos(datos) { const selectProceso = document.getElementById('filtro-proceso'); // Ajusta ID del select if (!selectProceso) return; // Extraer procesos únicos ignorando nulos const procesosUnicos = [...new Set(datos.map(d => d.PROCESO).filter(Boolean))].sort(); selectProceso.innerHTML = ''; procesosUnicos.forEach(proc => { const option = document.createElement('option'); option.value = proc; option.textContent = proc; selectProceso.appendChild(option); }); } // Llenar Dropdown de Auditores solo con el equipo activo function poblarFiltroAuditores() { const selectAuditor = document.getElementById('filtro-auditor'); // Ajusta ID del select if (!selectAuditor) return; selectAuditor.innerHTML = ''; AUDITORES_ACTIVOS.forEach(aud => { const option = document.createElement('option'); option.value = aud.nombre; option.textContent = aud.nombre; selectAuditor.appendChild(option); }); } // Actualizar tarjetas de KPI con datos reales de la lista filtrada function actualizarKpis(datos) { const totalAuditorias = document.getElementById('kpi-total'); const totalAuditores = document.getElementById('kpi-auditores'); const totalProcesos = document.getElementById('kpi-procesos'); if (totalAuditorias) totalAuditorias.innerText = datos.length; if (totalAuditores) { const auditoresConDatos = new Set(datos.map(d => d.AUDITOR)); totalAuditores.innerText = auditoresConDatos.size; } if (totalProcesos) { const procesosPresentes = new Set(datos.map(d => d.PROCESO).filter(Boolean)); totalProcesos.innerText = procesosPresentes.size; } }
