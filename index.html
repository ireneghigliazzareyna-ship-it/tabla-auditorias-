// Función principal para procesar los datos de BigQuery sin filtrar a nadie function procesarDatosDashboard(datosRaw) { if (!datosRaw || !Array.isArray(datosRaw)) return; // 1. Renderizar la tabla directamente con los correos originales renderTabla(datosRaw); // 2. Llenar los filtros de forma 100% dinámica poblarFiltros(datosRaw); // 3. Actualizar los contadores superiores (KPIs) actualizarKpis(datosRaw); } // Renderizar la tabla (Grid) function renderTabla(datos) { const tbody = document.getElementById('grid-body'); // Revisa que coincida con el ID de tu if (!tbody) return; tbody.innerHTML = ''; datos.forEach(row => { const tr = document.createElement('tr'); tr.innerHTML = ` ${row.FECHA_HORA ? new Date(row.FECHA_HORA.value || row.FECHA_HORA).toLocaleString() : '-'} ${row.PROCESO || '-'} ${row.AUDITOR || '-'} ${row.ID || '-'} `; tbody.appendChild(tr); }); } // Poblar desplegables de Auditor y Proceso automáticamente function poblarFiltros(datos) { const selectAuditor = document.getElementById('filtro-auditor'); // Ajusta ID según tu HTML const selectProceso = document.getElementById('filtro-proceso'); // Ajusta ID según tu HTML // Extraer valores únicos sin duplicados const auditoresUnicos = [...new Set(datos.map(d => d.AUDITOR).filter(Boolean))].sort(); const procesosUnicos = [...new Set(datos.map(d => d.PROCESO).filter(Boolean))].sort(); // Llenar Filtro de Auditores if (selectAuditor) { selectAuditor.innerHTML = ''; auditoresUnicos.forEach(auditor => { const option = document.createElement('option'); option.value = auditor; option.textContent = auditor; selectAuditor.appendChild(option); }); } // Llenar Filtro de Procesos if (selectProceso) { selectProceso.innerHTML = ''; procesosUnicos.forEach(proceso => { const option = document.createElement('option'); option.value = proceso; option.textContent = proceso; selectProceso.appendChild(option); }); } } // Actualizar contadores superiores function actualizarKpis(datos) { const totalAuditorias = document.getElementById('kpi-total'); const totalAuditores = document.getElementById('kpi-auditores'); const totalProcesos = document.getElementById('kpi-procesos'); if (totalAuditorias) totalAuditorias.innerText = datos.length; if (totalAuditores) { const auditoresPresentes = new Set(datos.map(d => d.AUDITOR).filter(Boolean)); totalAuditores.innerText = auditoresPresentes.size; } if (totalProcesos) { const procesosPresentes = new Set(datos.map(d => d.PROCESO).filter(Boolean)); totalProcesos.innerText = procesosPresentes.size; } }
