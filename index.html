// Función principal para recibir los datos de Verdi / API function procesarDatosDashboard(datosRaw) { if (!datosRaw || !Array.isArray(datosRaw)) { console.error("Los datos recibidos no son un arreglo válido:", datosRaw); return; } // 1. Mostrar las filas en la tabla renderTabla(datosRaw); // 2. Cargar opciones en los selectores desplegables poblarFiltros(datosRaw); // 3. Actualizar tarjetas superiores actualizarKpis(datosRaw); } // 1. Render de la Tabla (Grid) function renderTabla(datos) { const tbody = document.getElementById('grid-body'); // Asegúrate que tu en HTML tenga id="grid-body" if (!tbody) return; tbody.innerHTML = ''; datos.forEach(row => { // Extraer el auditor (si viene correo o nombre, lo toma tal cual) const auditor = row.AUDITOR || row.auditor || 'N/A'; const proceso = row.PROCESO || row.proceso || '-'; const idTransaccion = row.ID || row.id || '-'; // Manejar la fecha let fecha = '-'; if (row.FECHA_HORA) { const f = row.FECHA_HORA.value || row.FECHA_HORA; fecha = new Date(f).toLocaleString(); } const tr = document.createElement('tr'); tr.innerHTML = ` ${fecha} ${proceso} ${auditor} ${idTransaccion} `; tbody.appendChild(tr); }); } // 2. Poblar desplegables de PROCESO y AUDITOR function poblarFiltros(datos) { const selectAuditor = document.getElementById('filtro-auditor'); const selectProceso = document.getElementById('filtro-proceso'); // Obtener valores únicos const auditoresUnicos = [...new Set(datos.map(d => d.AUDITOR || d.auditor).filter(Boolean))].sort(); const procesosUnicos = [...new Set(datos.map(d => d.PROCESO || d.proceso).filter(Boolean))].sort(); // Cargar filtro de Auditores if (selectAuditor) { selectAuditor.innerHTML = ''; auditoresUnicos.forEach(aud => { const opt = document.createElement('option'); opt.value = aud; opt.textContent = aud; selectAuditor.appendChild(opt); }); } // Cargar filtro de Procesos (TMWH HU, OUT, etc.) if (selectProceso) { selectProceso.innerHTML = ''; procesosUnicos.forEach(proc => { const opt = document.createElement('option'); opt.value = proc; opt.textContent = proc; selectProceso.appendChild(opt); }); } } // 3. Actualizar KPIs function actualizarKpis(datos) { const elemTotal = document.getElementById('kpi-total'); const elemAuditores = document.getElementById('kpi-auditores'); const elemProcesos = document.getElementById('kpi-procesos'); if (elemTotal) elemTotal.innerText = datos.length; if (elemAuditores) { const auditoresUnicos = new Set(datos.map(d => d.AUDITOR || d.auditor).filter(Boolean)); elemAuditores.innerText = auditoresUnicos.size; } if (elemProcesos) { const procesosUnicos = new Set(datos.map(d => d.PROCESO || d.proceso).filter(Boolean)); elemProcesos.innerText = procesosUnicos.size; } }
